CREATE TABLE `logbook`.`allowedoauthclients` (
  `ClientID` VARCHAR(45) NOT NULL COMMENT 'Unique ID for the client',
  `ClientSecret` VARCHAR(255) NOT NULL COMMENT 'Secret for the client',
  `ClientName` VARCHAR(255) NOT NULL COMMENT 'Name for the client',
  `Callback` VARCHAR(255) NOT NULL COMMENT 'Allowed URL callback for the client',
  `ClientType` INT NOT NULL COMMENT 'Type of client (DotNetOpenAuth.OAuth2.ClientType.Public enum)',
  `Scopes` VARCHAR(255) NOT NULL COMMENT 'Allowed scopes for the client.',
  PRIMARY KEY (`ClientID`),
  UNIQUE INDEX `ClientID_UNIQUE` (`ClientID` ASC))
COMMENT = 'Identifies allowed oAuth client services';
