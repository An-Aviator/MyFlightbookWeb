CREATE TABLE `logbook`.`MaintenanceLog` (
  `id` INTEGER UNSIGNED NOT NULL AUTO_INCREMENT COMMENT 'Row ID',
  `idAircraft` INTEGER UNSIGNED NOT NULL DEFAULT 0 COMMENT 'Aircraft for which a change was made',
  `Description` VARCHAR(255) NOT NULL DEFAULT '' COMMENT 'What was done',
  `Date` DATETIME NOT NULL DEFAULT 0 COMMENT 'The date of the change',
  `User` VARCHAR(45) NOT NULL DEFAULT '' COMMENT 'The name of the user who made the change',
  `Comment` VARCHAR(255) COMMENT 'Any comment',
  PRIMARY KEY(`id`)
)
ENGINE = InnoDB
COMMENT = 'Tracks who did what w.r.t. maintenance on an airplane';
