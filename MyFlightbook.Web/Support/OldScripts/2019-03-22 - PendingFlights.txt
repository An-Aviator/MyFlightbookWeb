CREATE TABLE `pendingflights` (
  `id` varchar(255) NOT NULL COMMENT 'Key from the original source, or autogenerated, if appropriate.',
  `username` varchar(255) NOT NULL,
  `jsonflight` text NOT NULL,
  PRIMARY KEY (`id`),
  KEY `pflightuser_idx` (`username`),
  CONSTRAINT `pflightuser` FOREIGN KEY (`username`) REFERENCES `users` (`Username`) ON DELETE CASCADE ON UPDATE NO ACTION
) ENGINE=InnoDB DEFAULT CHARSET=utf8 COMMENT='Flights that are pending addition to a user''s logbook.';
