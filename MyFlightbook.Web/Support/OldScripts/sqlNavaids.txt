CREATE TABLE `logbook`.`NavaidTypes` (
  `Code` VARCHAR(5) NOT NULL DEFAULT '' COMMENT 'The code from the FAA data',
  `FriendlyName` VARCHAR(45) NOT NULL DEFAULT '',
  PRIMARY KEY(`Code`)
)
ENGINE = InnoDB
COMMENT = 'Types of navaids';

load data local infile 'C:\\NavaidTable.csv'
    ignore
    into table navaidtypes
    fields optionally enclosed by '"' terminated by ','
    lines terminated by '\r\n'
ignore 1 lines

-- Support for user-created navaids/waypoints
ALTER TABLE `logbook`.`airports` MODIFY COLUMN `AirportID` VARCHAR(10) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT 'ICAO or IATA code',
 MODIFY COLUMN `FacilityName` VARCHAR(128) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL COMMENT 'Friendly Name',
 MODIFY COLUMN `Lat` VARCHAR(15) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL DEFAULT 0 COMMENT 'Latitude',
 MODIFY COLUMN `Long` VARCHAR(15) CHARACTER SET utf8 COLLATE utf8_general_ci NOT NULL DEFAULT 0 COMMENT 'Longitude',
