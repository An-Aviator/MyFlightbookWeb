CREATE TABLE `logbook`.`basicmedevent` (
  `idBasicMedEvent` INT NOT NULL AUTO_INCREMENT COMMENT 'Primary key',
  `eventtype` INT NULL COMMENT 'Indicates type of event - medical course or physician visit',
  `eventdate` DATE NOT NULL COMMENT 'Date of event',
  `username` VARCHAR(255) NOT NULL COMMENT 'Username associated with this event',
  `description` VARCHAR(255) NULL COMMENT 'Description of the event',
  PRIMARY KEY (`idBasicMedEvent`),
  INDEX `users` (`username` ASC),
  CONSTRAINT `fkusername`
    FOREIGN KEY (`username`)
    REFERENCES `logbook`.`users` (`Username`)
    ON DELETE CASCADE
    ON UPDATE NO ACTION)
COMMENT = 'Contains basicmed events for users';
